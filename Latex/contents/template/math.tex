\subsection{数学符号和公式}


\begin{equation}
f(x) = x^2 + 1
\end{equation}

\begin{equation}
\frac{2h}{\pi}\int_{0}^{\infty}\frac{\sin\left( \omega\delta \right)}{\omega}
		\cos\left( \omega x \right) \dd\omega = 
        \begin{cases}
                h, \ \left| x \right| < \delta,\\ 
                \frac{h}{2}, \ x = \pm \delta,\\ 
                0, \ \left| x \right| > \delta.
        \end{cases}
\end{equation}

\begin{equation}

        y= \begin{cases}
                -x,\quad &x\leq 0 \\
                x,\quad &x>0
            \end{cases} 
        \end{aligned}
\end{equation}
